## 树的遍历主要有以下三种方式：先序遍历、中序遍历和后序遍历

#### 递归实现（递归实现比较简单）

```c
// 树的先序遍历
void find(Node* root) {
  if (root == NULL) {
    return;
  }
  printf("%d", root->data);
  
  find(root->left);
  find(root->right);
}

// 树的中序遍历
void find(Node* root) {
  if (root == NULL) {
    return;
  }
  find(root->left);
  
  printf("%d", root->data);
  
  find(root->right);
}

// 树的后序遍历
void find(Node* root) {
  if (root == NULL) {
    return;
  }
  find(root->left);
  find(root->right);
  
  printf("%d", root->data);
}
```

## 层序遍历（用队列+孩子兄弟表示法）

```c
#include <stdio.h>
#include <stdlib.h>

// 队列
// 顺序队列
/* 孩子兄弟表示法 */
typedef struct childBro {
	int data;
	struct childBro* child; // 孩子指针
	struct childBro* sibling; // 兄弟指针
}Node;
int* queue; // 存储数据的数组
int front, end; // 队头尾指针
int maxSize; // 最大容量
// 全局声明根结点
Node* root;
Node* tempNode;

// 初始化队列
void initQueue() {
    queue = (int*)malloc(sizeof(int) * 100);
    if (queue == NULL) {
        printf("无法分配内存！！");
        exit(0);
    }
    front = end = 0;
    maxSize = 100;
}

// 判断队列是否已满
int isFull() {
    if ((end + 1) % maxSize == front) {
        return 1;
    }
    else {
        return 0;
    }
}

// 判断队列为空
int isEmpty() {
    if (end == front) {
        return 1;
    }
    else {
        return 0;
    }
}

// 入队
void insert_queue(int key) {
    if (isFull()) {
        printf("队满，无法插入!!");
        return;
    }
    end = (end + 1) % maxSize;
    queue[end] = key;
}

// 出队
int delete_queue() {
    if (isEmpty()) {
        printf("队列为空，无法完成出队操作!!");
		exit(0);
    }
    front = (front + 1) % maxSize;
	printf("%d ", queue[front]);
	return queue[front];
}


// 孩子兄弟表示法



// 初始化  相当于创建根结点
void init(int key) {
	root = (Node*)malloc(sizeof(Node)); // 先分配空间
	if (root == NULL) {
		printf("无法分配内存!\n");
		return;
	}

	root->data = key;
	root->child = NULL;
	root->sibling = NULL;
}

// 查找节点，即遍历树，注意该递归有两个出口，为空和找到目标值
Node* find(Node* temp, int parent) {
	if (temp == NULL) {
		return NULL;
	}
	else {
		if (temp->data == parent) {
			return temp; // 找到父节点
		}
		else {
			Node* t;
			t = find(temp->child, parent);
			if (t != NULL) {
				return t;
			}
			t = find(temp->sibling, parent);
			if (t != NULL) {
				return t;
			}
		}
	}
}

// 插入节点
void insert(int key, int parent) {
	// 插入之前，先找到要插入的父节点在哪
	Node* temp = (Node*)malloc(sizeof(Node)); // 定义一个临时指针，用来查找父节点

	temp = find(root, parent);
	if (temp == NULL) {
		printf("无法分配内存!\n");
		return;
	}

	if (temp == NULL) {
		printf("没有该父节点!\n");
	}
	else {
		if (temp->child == NULL) { // 父节点没有孩子，则该节点应为孩子节点
			Node* node = (Node*)malloc(sizeof(Node)); // 定义一个临时指针，用来查找父节点
			if (node == NULL) {
				printf("无法分配内存!\n");
				return;
			}

			node->data = key;
			node->child = NULL;
			node->sibling = NULL;
			temp->child = node;
		}
		else { // 父节点有孩子节点，那么该节点应为孩子节点的兄弟节点
			temp = temp->child;
			Node* node = (Node*)malloc(sizeof(Node)); // 定义一个临时指针，用来查找父节点
			if (node == NULL) {
				printf("无法分配内存!\n");
				return;
			}

			node->data = key;
			node->child = NULL;
			node->sibling = NULL;
			temp->sibling = node;
		}
	}
}


// 根据出队的值，找到其节点
void get_node(Node* node, int key) {
	if (node->data == key) {
		tempNode = node;
	}
	if (node->sibling != NULL) {
		get_node(node->sibling, key);
	}
	if (node->child != NULL) {
		get_node(node->child, key);
	}
}

// 层序遍历
void levelOrder(Node* root) {
	if (root != NULL) {
		insert_queue(root->data);
	}
	else {
		printf("该树为空!\n");
		return;
	}

	while (!isEmpty()) {
		int temp = delete_queue(); // 元素出队
		get_node(root, temp); // 找点当前元素的节点
		if (tempNode->child != NULL) { // 每次入队当前节点的孩子 和 当前节点孩子的兄弟节点
			insert_queue(tempNode->child->data);
            if (tempNode->child->sibling != NULL) {
                insert_queue(tempNode->child->sibling->data);
            }
		}
	}
}

// 验证结果
int main() {
	initQueue();
	init(1);
	insert(2, 1);
	insert(3, 1);
	insert(4, 2);
	insert(5, 2);
	insert(6, 3);
	insert(7, 3);
	levelOrder(root);
}
```
