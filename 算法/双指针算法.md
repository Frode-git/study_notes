#### 双指针算法

> 核心思想：将双层循环的时间复杂度优化到 O(n)

```c++
// 朴素算法：
for (int i = 0; i < n; i ++) {
  for (int j = i; j < n; j ++) {
    // 每道题的具体逻辑
    // O(n ^ 2)
  }
}

// 双指针算法：
for (int i = 0; i < n; i ++) {
  int j = i;
  while (j < n && check(i, j)) j ++;
  
  // 每道题的具体逻辑
}
```



### 最长连续不重复子序列

> 例： 1 2 2 3 5  =>   2 3 5  ，长度为 3

```c++
#include <iostream>
using namespace std;

const int N = 100010;

int a[N], s[N];
int n;

int main() {
    cin >> n;
    for (int i = 0; i < n; i ++) cin >> a[i];
    
    int res = 0;
    for (int i = 0, j = 0; i < n; i ++) {
        s[a[i]] ++; // 每次遍历到一个数字，加入
        while (s[a[i]] > 1) { // 如果当前这个数的个数大于1，表明存在重复数字，将j后移，并把之前j指向的数踢掉
            s[a[j]] --;
            j ++;
        }
        res = max(res, i - j + 1); // 维护最大长度
    }
    
    cout << res << endl;
    
    return 0;
}
```



### 数组元素的目标和

> 两个升序数组，求一组数对(i, j)，使得a[i] + b[j] = x

```c++
#include <iostream>
using namespace std;

const int N = 100010;
int m, n, x;
int a[N], b[N];

int main() {
    scanf("%d%d%d", &n, &m, &x);
    for (int i = 0; i < n; i ++) scanf("%d", &a[i]);
    for (int i = 0; i < m; i ++) scanf("%d", &b[i]);
    
    for (int i = 0, j = m - 1; i < n; i ++) {
        while(j >= 0 && a[i] + b[j] > x) j --;  // 对于每一个i，求最小的一个j，使得a[i] + b[j] > x,然后让j左移，这样，i右移，j必须左移，即具有单调性
        if (a[i] + b[j] == x) {
            printf("%d %d\n", i, j);
            break;
        }
    }
    
    return 0;
}
```



### 判断子序列

> 序列a：a1, a3, a5    序列b：a1, a2, a3, a4, a5, a6,a7   序列a是序列b的子序列

```c++
#include <iostream>
#include <cstring>

using namespace std;

const int N = 100010;
int n, m;
int a[N], b[N];

int main() {
    scanf("%d%d", &n, &m);
    
    for(int i = 0; i < n; i ++) scanf("%d", &a[i]);
    for(int i = 0; i < m; i ++) scanf("%d", &b[i]);
    
    int i = 0, j = 0;
    while(i < n && j < m) { // 时间复杂度是 O(n)
        if (a[i] == b[j]) i ++;
        j ++;
    }
    
    if (i == n) puts("Yes");
    else puts("No");
    
    return 0;
}
```
